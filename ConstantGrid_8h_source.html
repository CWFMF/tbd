172 [[nodiscard]] static ConstantGrid<T, V>* readTiff(const string& filename,
173 TIFF* tif,
174 GTIF* gtif,
175 const topo::Point& point,
176 std::function<T(V, V)> convert)
177 {
178 logging::info("Reading file %s", filename.c_str());
179#ifdef DEBUG_GRIDS
180 // auto min_value = std::numeric_limits<int16_t>::max();
181 // auto max_value = std::numeric_limits<int16_t>::min();
182 auto min_value = std::numeric_limits<V>::max();
183 auto max_value = std::numeric_limits<V>::min();
184#endif
185 logging::debug("Reading a raster where T = %s, V = %s",
186 typeid(T).name(),
187 typeid(V).name());
188 logging::debug("Raster type V has limits %ld, %ld",
189 std::numeric_limits<V>::min(),
190 std::numeric_limits<V>::max());
191 const GridBase grid_info = read_header(tif, gtif);
192 int tile_width;
193 int tile_length;
194 TIFFGetField(tif, TIFFTAG_TILEWIDTH, &tile_width);
195 TIFFGetField(tif, TIFFTAG_TILELENGTH, &tile_length);
196 void* data;
197 uint32_t count;
198 TIFFGetField(tif, TIFFTAG_GDAL_NODATA, &count, &data);
199 logging::check_fatal(0 == count, "NODATA value is not set in input");
200 logging::debug("NODATA value is '%s'", static_cast<char*>(data));
201 const auto nodata_orig = stoi(string(static_cast<char*>(data)));
202 logging::debug("NODATA value is originally parsed as %d", nodata_orig);
203 const auto nodata_input = static_cast<V>(stoi(string(static_cast<char*>(data))));
204 logging::debug("NODATA value is parsed as %d", nodata_input);
205 auto actual_rows = grid_info.calculateRows();
206 auto actual_columns = grid_info.calculateColumns();
207 const auto coordinates = grid_info.findFullCoordinates(point, true);
208 auto min_column = max(static_cast<FullIdx>(0),
209 static_cast<FullIdx>(std::get<1>(*coordinates) - static_cast<FullIdx>(MAX_COLUMNS) / static_cast<FullIdx>(2)));
210 if (min_column + MAX_COLUMNS >= actual_columns)
211 {
212 min_column = max(static_cast<FullIdx>(0), actual_columns - MAX_COLUMNS);
213 }
214 // make sure we're at the start of a tile
215 const auto tile_column = tile_width * static_cast<FullIdx>(min_column / tile_width);
216 const auto max_column = static_cast<FullIdx>(min(min_column + MAX_COLUMNS - 1, actual_columns));
217#ifdef DEBUG_GRIDS
218 logging::check_fatal(min_column < 0, "Column can't be less than 0");
219 logging::check_fatal(max_column - min_column > MAX_COLUMNS, "Can't have more than %d columns", MAX_COLUMNS);
220 logging::check_fatal(max_column > actual_columns, "Can't have more than actual %d columns", actual_columns);
221#endif
222 auto min_row = max(static_cast<FullIdx>(0),
223 static_cast<FullIdx>(std::get<0>(*coordinates) - static_cast<FullIdx>(MAX_ROWS) / static_cast<FullIdx>(2)));
224 if (min_row + MAX_COLUMNS >= actual_rows)
225 {
226 min_row = max(static_cast<FullIdx>(0), actual_rows - MAX_ROWS);
227 }
228 const auto tile_row = tile_width * static_cast<FullIdx>(min_row / tile_width);
229 const auto max_row = static_cast<FullIdx>(min(min_row + MAX_ROWS - 1, actual_rows));
230#ifdef DEBUG_GRIDS
231 logging::check_fatal(min_row < 0, "Row can't be less than 0 but is %d", min_row);
232 logging::check_fatal(max_row - min_row > MAX_ROWS, "Can't have more than %d rows but have %d", MAX_ROWS, max_row - min_row);
233 logging::check_fatal(max_row > actual_rows, "Can't have more than actual %d rows", actual_rows);
234#endif
235 T nodata_value = convert(nodata_input, nodata_input);
236 logging::check_fatal(
237 convert(nodata_input, nodata_input) != nodata_value,
238 "Expected nodata value to be returned from convert()");
239 vector<T> values(static_cast<size_t>(MAX_ROWS) * MAX_COLUMNS, nodata_value);
240 logging::verbose("%s: malloc start", filename.c_str());
241 int bps = std::numeric_limits<V>::digits + (1 * std::numeric_limits<V>::is_signed);
242 int bps_file;
243 TIFFGetField(tif, TIFFTAG_BITSPERSAMPLE, &bps_file);
244 logging::check_fatal(bps != bps_file,
245 "Raster %s type is not expected type (%ld bits instead of %ld)",
246 filename.c_str(),
247 bps_file,
248 bps);
249#ifdef DEBUG_GRIDS
250 int bps_int16_t = std::numeric_limits<int16_t>::digits + (1 * std::numeric_limits<int16_t>::is_signed);
251 logging::debug("Size of pointer to int is %ld vs %ld", sizeof(int16_t*), sizeof(V*));
252 logging::debug("Raster %s calculated bps for type V is %ld; tif says bps is %ld; int16_t is %ld",
253 filename.c_str(),
254 bps,
255 bps_file,
256 bps_int16_t);
257#endif
258 const auto tile_size = TIFFTileSize(tif);
259 logging::debug("Tile size for reading %s is %ld", filename.c_str(), tile_size);
260 const auto buf = _TIFFmalloc(tile_size);
261 logging::verbose("%s: read start", filename.c_str());
262 const tsample_t smp{};
263 logging::debug("Want to clip grid to (%d, %d) => (%d, %d) for a %dx%d raster",
264 min_row,
265 min_column,
266 max_row,
267 max_column,
268 actual_rows,
269 actual_columns);
270 for (auto h = tile_row; h <= max_row; h += tile_length)
271 {
272 for (auto w = tile_column; w <= max_column; w += tile_width)
273 {
274 TIFFReadTile(tif, buf, static_cast<uint32_t>(w), static_cast<uint32_t>(h), 0, smp);
275 for (auto y = 0; (y < tile_length) && (y + h <= max_row); ++y)
276 {
277 // read in so that (0, 0) has a hash of 0
278 const auto y_row = static_cast<HashSize>((h - min_row) + y);
279 const auto actual_row = (max_row - min_row) - y_row;
280 if (actual_row >= 0 && actual_row < MAX_ROWS)
281 {
282 for (auto x = 0; (x < tile_width) && (x + w <= max_column); ++x)
283 {
284 const auto offset = y * tile_width + x;
285 const auto actual_column = ((w - min_column) + x);
286 if (actual_column >= 0 && actual_column < MAX_ROWS)
287 {
288 const auto cur_hash = actual_row * MAX_COLUMNS + actual_column;
289 auto cur = *(static_cast<V*>(buf) + offset);
290#ifdef DEBUG_GRIDS
291 min_value = min(cur, min_value);
292 max_value = max(cur, max_value);
293#endif
294 // try
295 // {
296 values.at(cur_hash) = convert(cur, nodata_input);
297 // logging::check_fatal(Settings::fuelLookup().values.at(cur_hash));
298 // }
299 // // catch (const std::out_of_range& err)
300 // catch (const std::exception& err)
301 // {
302 // logging::error("Error trying to read tiff");
303 // logging::debug("cur = %d", cur);
304 // logging::debug("nodata_input = %d", nodata_input);
305 // logging::debug("T = %s, V = %s", typeid(T).name(), typeid(V).name());
306 // if constexpr (std::is_same_v<T, const tbd::fuel::FuelType*>)
307 // {
308 // auto f = static_cast<const tbd::fuel::FuelType*>(values.at(cur_hash));
309 // logging::debug("fuel %s has code %d",
310 // tbd::fuel::FuelType::safeName(f),
311 // tbd::fuel::FuelType::safeCode(f));
312 // }
313 // logging::warning(err.what());
314 // // logging::fatal(err.what());
315 // tbd::sim::Settings::fuelLookup().listFuels();
316 // throw err;
317 // }
318 }
319 }
320 }
321 }
322 }
323 }
324 logging::verbose("%s: read end", filename.c_str());
325 _TIFFfree(buf);
326 logging::verbose("%s: free end", filename.c_str());
327 const auto new_xll = grid_info.xllcorner() + (static_cast<double>(min_column) * grid_info.cellSize());
328 const auto new_yll = grid_info.yllcorner()
329 + (static_cast<double>(actual_rows) - static_cast<double>(max_row))
330 * grid_info.cellSize();
331#ifdef DEBUG_GRIDS
332 logging::check_fatal(new_yll < grid_info.yllcorner(),
333 "New yllcorner is outside original grid");
334#endif
335 logging::verbose("Translated lower left is (%f, %f) from (%f, %f)",
336 new_xll,
337 new_yll,
338 grid_info.xllcorner(),
339 grid_info.yllcorner());
340 const auto num_rows = max_row - min_row + 1;
341 const auto num_columns = max_column - min_column + 1;
342 auto result = new ConstantGrid<T, V>(grid_info.cellSize(),
343 num_rows,
344 num_columns,
345 nodata_input,
346 nodata_value,
347 new_xll,
348 new_yll,
349 new_xll + (static_cast<double>(num_columns) + 1) * grid_info.cellSize(),
350 new_yll + (static_cast<double>(num_rows) + 1) * grid_info.cellSize(),
351 string(grid_info.proj4()),
352 std::move(values));
353 auto new_location = result->findCoordinates(point, true);
354#ifdef DEBUG_GRIDS
355 logging::check_fatal(nullptr == new_location, "Invalid location after reading");
356#endif
357 logging::note("Coordinates are (%d, %d => %f, %f)",
358 std::get<0>(*new_location),
359 std::get<1>(*new_location),
360 std::get<0>(*new_location) + std::get<2>(*new_location) / 1000.0,
361 std::get<1>(*new_location) + std::get<3>(*new_location) / 1000.0);
362#ifdef DEBUG_GRIDS
363 logging::note("Values for %s range from %d to %d",
364 filename.c_str(),
365 min_value,
366 max_value);
367#endif
368 return result;
369 }
